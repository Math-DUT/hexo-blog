解析：

完全二叉树的一个重要性子：对于有 NNN 个节点的完全二叉树，如果我们用数组存储（根节点下标为 1），则对于任意节点 iii：

  • 左孩子下标为 2×i2 \times i2×i
  • 右孩子下标为 2×i+12 \times i + 12×i+1

解题思路：

  1. 完全二叉树的后序遍历序列中，最后一个元素是根节点
  2. 我们可以递归地重建树：对于区间 [l,r][l, r][l,r]，其中点左侧是左子树，右侧是右子树，最后是根节点
  3. 使用 DFS 从后序遍历序列重建树，并将结果存入数组（层序存储）
  4. 最后按数组顺序输出即可

具体实现时，我们用一个 tree 数组存储层序遍历结果，通过后序遍历的顺序递归填充。

代码：

#include <iostream>
#include <vector>
using namespace std;

int n;

vector<int> postorder;

vector<int> tree;


// idx: 当前子树根节点在 tree 中的位置

// l, r: 当前子树在后序遍历中的区间 [l, r]

void dfs(int idx, int l, int r) {

if (l > r) return;


// 后序遍历的最后一个元素是当前子树的根
tree[idx] = postorder[r];

// 计算左右子树的大小
int leftSize = (r - l + 1) / 2;
int rightSize = (r - l + 1) - leftSize - 1;

// 递归构建左右子树
dfs(idx * 2, l, l + leftSize - 1);           // 左子树
dfs(idx * 2 + 1, l + leftSize, r - 1);       // 右子树



}


int main() {

cin >> n;

postorder.resize(n);

tree.resize(n + 1);


for (int i = 0; i < n; i++) {
    cin >> postorder[i];
}

dfs(1, 0, n - 1);

// 输出层序遍历结果
for (int i = 1; i <= n; i++) {
    cout << tree[i] << (i == n ? "" : " ");
}
cout << endl;

return 0;



}
